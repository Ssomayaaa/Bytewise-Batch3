-- Query to select RNA structures with length greater than 12 and added after December 31, 2008

select * from rna where len>12 and timestamp> '2008-12-31 00:00:00' limit 1000;

-- Query to count the number of pre-computed RNA records that are active and were last updated before January 1, 2022

select * from rnc_rna_precomputed where is_active = TRUE and update_date < '2022-01-01' limit 1000;

-- Query to count the total number of pre-computed RNA records for snoRNA and tRNA recorded in 2011, 2014, 2016, and 2020

select * from rnc_rna_precomputed where rna_type in ('snoRNA', 'tRNA') and extract (year from update_date) IN (2011, 2014, 2016, 2020) limit 1000;

-- Query to select names of all databases built for RNA with a minimum length other than 100, 200, 300, 400, and 15

select display_name from rnc_database where min_length not in (100, 200, 300, 400, 15) limit 1000;

-- Query to select 500 records of sequences for active regions, renaming the region_name column to myregions

select region_name as myregions from rnc_sequence_regions limit 500;
